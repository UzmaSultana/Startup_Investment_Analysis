import seaborn as sns

if 'deal_equity' in df.columns and 'ask_equity' in df.columns:
    # Create the equity_diff column
    df['equity_diff'] = df['deal_equity'] - df['ask_equity']

    sns.boxplot(x=df['equity_diff'], color='salmon')
